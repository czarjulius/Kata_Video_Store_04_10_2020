113       def self.replace_escaped_symbols(text)
 
114         copy_of_text = String.new(text)
 
115 
116         copy_of_text.gsub!(/\|/, "||")
 
117 
118         copy_of_text.gsub!(/'/, "|'")
 
119         copy_of_text.gsub!(/\n/, "|n")
 
120         copy_of_text.gsub!(/\r/, "|r")
 
121         copy_of_text.gsub!(/\]/, "|]")
 
122         
123         copy_of_text
 
124       end

146       def self.convert_time_to_java_simple_date(time)
 
147         gmt_offset = time.gmt_offset
 
148         gmt_sign = gmt_offset < 0 ? "-" : "+"
 
149         gmt_hours = gmt_offset.abs / 3600
 
150         gmt_minutes = gmt_offset.abs % 3600
 
151 
152         millisec = time.usec == 0 ? 0 : time.usec / 1000
 
153 
154         #Time string in Java SimpleDateFormat
 
155         sprintf("#{time.strftime("%Y-%m-%d'T'%H:%M:%S.")}%03d#{gmt_sign}%02d%02d", millisec, gmt_hours, gmt_minutes)
